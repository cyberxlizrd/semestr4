\subsection*{Výpočet střední hodnoty zpoždění efektu Chorus}

Pro výpočet zpoždění $\tau$:
\begin{equation}
\tau =  \frac{k - 0,5}{\tau} = \frac{2k - 1}{2f_{\min}}
\end{equation}

\begin{enumerate}
    \item Pro 4. minimum ($k=4, f_{min_4}=514$ Hz): 
    $\tau_1 = \frac{2 \cdot 4 - 1}{2 \cdot 514} = \frac{7}{1028} \approx 6,81 \text{ ms}$
    
    \item Pro 5. minimum ($k=5, f_{min_5}=645$ Hz): 
    $\tau_2 = \frac{2 \cdot 5 - 1}{2 \cdot 645} = \frac{9}{1290} \approx 6,98 \text{ ms}$
    
    \item Pro 6. minimum ($k=6, f_{min_6}=765$ Hz):
    $\tau_3 = \frac{2 \cdot 6 - 1}{2 \cdot 765} = \frac{11}{1530} \approx 7,19 \text{ ms}$
\end{enumerate}

\textbf{Průměrná hodnota:}
\begin{equation}
\tau_{avg} = \frac{6,81 + 6,98 + 7,19}{3} \approx \mathbf{6,99 \text{ ms}}
\end{equation}
